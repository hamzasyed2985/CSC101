#include <iostream>
using namespace std;

int main()
{
    int size;
    int count = 1;
    cout << "Enter size of array : ";
    cin >> size;
    int num[size];
    for (int i = 0; i < size; i++)
    {
        cout << "Enter the next number :";
        cin >> num[i];
    }

    for (int i = 0; i < size; i++)
    {

        if (num[i] == num[i + 1])
        {
            count++;
        }
        else if (num[i + 1] == num[i + 2])
            count = 1;
    }
    cout << count;
}
